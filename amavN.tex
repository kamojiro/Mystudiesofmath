\section{Amalgamated free product von Neumann algevras}
\cite{ueda1999amalgamated}
\begin{theorem}
  For each n.s.f.f. weight $\varphi$ on $N$, we have
  \begin{align*}
    \sigma_t^{\varphi \circ E} = \underset{s \in S}{*_N} \sigma_t^{\varphi \circ E_s} \; for \; (t \in \R).
  \end{align*}
\end{theorem}
